49bool GUIEditBox::show(uint16_t a_x, uint16_t a_y, char* a_pText, size_t a_size,EditMode a_mode)
50{
51 mode = a_mode;
52 if(mode == MODE_NUMPADOVERWRITE) {
53 isInsert = false;
54 edtCursor = 0;
55 psk->showNumPad(ptft->height() - psk->KB_HEIGHT); // キーボードを表示
56 } else {
57 edtCursor = strlen(a_pText);
58 psk->show(ptft->height() - psk->KB_HEIGHT); // キーボードを表示
59 isInsert = true;
60 }
61 p = a_pText;
62 edtX = a_x;
63 edtY = a_y;
64
65 int waitCnt = 0;
66 bool retReason = false;
67
68 while (true) {
69 uint8_t ret = psk->checkTouch();
70 if (ret == 0xff) { // タッチされていない場合はカーソルを点滅させたりする
71 waitCnt++;
72 waitCnt = waitCnt % 0xFF;
73 dispCursor(waitCnt, false); // カーソルを点滅させる
74 delay(5); // 少し待つ
75 continue;
76 } else {
77 dispCursor(waitCnt, true); // カーソルを消す
78 waitCnt = 0; // カーソルの点滅をリセット
79 }
80 // 特殊コードの処理
81 if (ret == 0x0d) {
82 retReason = true;
83 break; // Enterキーが押されたら終了
84 } else if (ret == 0x1b) { // ESCキーが押されたらキャンセル
85 retReason = false;
86 break;
87 } else if (ret == 0x13) { // ↚キー
88 edtCursor--;
89 if (edtCursor < 0) {
90 edtCursor = 0; // カーソルが先頭を超えないようにする
91 }
92 if (edtCursor > strlen(p)) {
93 edtCursor = strlen(p); // カーソルが文字列の末尾を超えないようにする
94 }
95 ret = 0;
96 } else if (ret == 0x14) { // ⇒キー
97 if (p[edtCursor] == 0) {
98 continue;
99 }
100 edtCursor++;
101 ret = 0;
102 } else if (ret == 0x7e) {
103 isInsert = !isInsert;
104 ret = 0;
105 }
106 if (ret == 0) {
107 continue;
108 }
109 // 一般文字の処理
110 //if (strlen(p) >= a_size - 1) {
111 // continue; // 文字列が最大長に達している場合は何もしない
112 //}
113
114 if (ret == 0x08) { // Backspaceキーが押されたら削除
115 if (edtCursor > 0) {
116 for (size_t i = edtCursor - 1; i < strlen(p); i++) {
117 p[i] = p[i + 1]; // 文字を左にシフト
118 }
119 edtCursor--;
120 } else {
121 continue; // カーソルが先頭にある場合は何もしない
122 }
123 } else if (ret == 0x7F) { // DELETキーが押されたらその場の文字を削除
124 if (p[edtCursor] == 0) {
125 continue; // カーソルが末尾にある場合は何もしない
126 }
127 if (edtCursor < strlen(p)) { // Deleteキーが押されたら削除
128 for (size_t i = edtCursor; i < strlen(p); i++) {
129 p[i] = p[i + 1]; // 文字を左にシフト
130 }
131 }
132 } else if (p[edtCursor] == '\0') { // 普通の文字の場合、カーソルが末尾にあるときは追加
133 if (edtCursor >= a_size) {
134 continue; // 文字列が最大長に達している場合は何もしない
135 }
136 p[edtCursor++] = ret;
137 p[edtCursor] = '\0'; // 文字列の終端を設定
138 } else {
139 if (isInsert) { // 挿入モード
140 if (strlen(p) >= a_size - 1) {
141 continue; // 文字列が最大長に達している場合は何もしない
142 } else if (edtCursor < strlen(p)) {
143 for (size_t i = strlen(p); i > edtCursor; i--) {
144 p[i] = p[i - 1]; // 文字を右にシフト
145 }
146 }
147 p[edtCursor++] = ret;
148 } else { // 上書きモード
149 p[edtCursor++] = ret;
150 if (edtCursor >= a_size - 1) {
151 edtCursor = a_size - 1; // カーソルが最大長を超えないようにする
152 }
153 }
154 }
155 // 文字の変化に伴う再描画
156 ptft->fillRect(edtX, edtY, 8*a_size, 16, STDCOLOR.BLACK); // 入力欄をクリア
157 ptft->setCursor(edtX, edtY);
158 ptft->printf(p);
159 }
160 return retReason; // trueならEnterキーが押された、falseならESCキーが押された
