125int ScreenKeyboard::getKeyFromTouch(int x, int y) const {
126 // キーボード領域外なら-1
127 if (x < 0 || x >= currWidth || y < 0 || y >= currHeight) return -1;
128 int col = x / currSize;
129 int row = y / currSize;
130 if (col >= currCols || row >= currRows) return -1;
131 return row * currCols + col;
132}

140uint8_t ScreenKeyboard::checkTouch() {
141 if (!m_ts) return 0xFF;
142 if (!m_ts->touched()) return 0xFF;
143 TS_Point p = m_ts->getPointOnScreen();
144 p.y = p.y - keyboardY;
145 if (p.y <0 || p.y >= currHeight) return 0xFF; // キーボード領域外なら-1
146 uint8_t code = 0xFF;
147 if (m_kbMode == KM_NUMPAD_1) {
148 code = npCodes[getKeyFromTouch(p.x, p.y)];
149 } else {
150 code = keycodes[m_kbType][getKeyFromTouch(p.x, p.y)];
151 }
152 while(m_ts->touched()) { // キーが離されてから初めて入力されたと判断する
153 delay(100);
154 }
155 if (code == 0x0f) {
156 m_kbType = static_cast<KBType>((m_kbType + 1) % 3);
157 show(keyboardY); // キーボードを再描画
158 return 0xFF;
159 }
160 return code;
161}
