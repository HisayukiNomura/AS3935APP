103bool InetAction::getHostByName(const char* hostname, int timeout_ms)
104{
105 isDNSFinished = 0; // DNS名前解決の完了フラグを初期化
106 ip4_addr_t addr;
107
108 err_t err = dns_gethostbyname(hostname, &addr, dns_callback, this);
109 if (err == ERR_OK) { // キャッシュヒット
110 addrSNTP.addr = addr.addr; // キャッシュから取得したIPアドレスを設定
111 return true;
112 } else if (err != ERR_INPROGRESS) {
113 dbgprintf("DNS ERROR: %s, (%d)\n", hostname, err);
114 return false; // 初期化に失敗
115 } else {
116 // 非同期の場合はポーリングで待機
117 while (timeout_ms > 0) {
118 sleep_ms(100); // 100ミリ秒待機
119 cyw43_arch_poll(); // lwIPイベント処理
120 if (isDNSFinished == 1) { // 成功
121 return true;
122 } else if (isDNSFinished == 2) {
123 dbgprintf("UNKNOWN ERROR: %s\n", hostname);
124 return false; // 失敗
125 }
126 timeout_ms -= 100;
127 }
128 dbgprintf("DNS TIMEOUT: %s\n", hostname);
129 return false;
130 }
131}

149extern "C" void setRTC(u32_t sec)
150{
151 time_t epoch = sec;
152 struct tm* t = gmtime(&epoch);
153 datetime_t dt = {
154 .year = (int16_t)(t->tm_year + 1900),
155 .month = (int8_t)(t->tm_mon + 1),
156 .day = (int8_t)t->tm_mday,
157 .dotw = (int8_t)t->tm_wday,
158 .hour = (int8_t)t->tm_hour,
159 .min = (int8_t)t->tm_min,
160 .sec = (int8_t)t->tm_sec};
161 bool bRet = rtc_set_datetime(&dt);
162 if (bRet == false) {
163 dbgprintf("rtc_set_datetime failed\n");
164 } else {
165 dbgprintf("rtc_set_datetime success: %04d/%02d/%02d %02d:%02d:%02d\n", dt.year, dt.month, dt.day, dt.hour, dt.min, dt.sec);
166 time_t now;
167 rtc_get_datetime(&dt);

169 t.tm_year = dt.year - 1900;
170 t.tm_mon = dt.month - 1;
171 t.tm_mday = dt.day;
172 t.tm_hour = dt.hour;
173 t.tm_min = dt.min;
174 t.tm_sec = dt.sec;
175 now = mktime(&t);
176 struct timeval tv;
177 tv.tv_sec = now;
178 tv.tv_usec = 0;
179 settimeofday(&tv, NULL);
180 setlocale(LC_TIME, "ja_JP.UTF-8");

195extern "C" void ntp_recv(void* arg, struct udp_pcb* pcb, struct pbuf* p, const ip_addr_t* addr, u16_t port)
196{
197 NTP_T* state = (NTP_T*)arg;
198 uint8_t mode = pbuf_get_at(p, 0) & 0x7;
199 uint8_t stratum = pbuf_get_at(p, 1);
200
201 // Check the result
202 if (ip_addr_cmp(addr, &state->ntp_server_address) && port == NTP_PORT && p->tot_len == NTP_MSG_LEN &&
203 mode == 0x4 && stratum != 0) {
204 uint8_t seconds_buf[4] = {0};
205 pbuf_copy_partial(p, seconds_buf, sizeof(seconds_buf), 40);
206 uint32_t seconds_since_1900 = seconds_buf[0] << 24 | seconds_buf[1] << 16 | seconds_buf[2] << 8 | seconds_buf[3];
207 uint32_t seconds_since_1970 = seconds_since_1900 - NTP_DELTA;
208 time_t epoch = seconds_since_1970;
209
210 // 結果の表示など
211 struct tm* utc = gmtime(&epoch);
212 // printf("got ntp response: %02d/%02d/%04d %02d:%02d:%02d\n", utc->tm_mday, utc->tm_mon + 1, utc->tm_year + 1900, utc->tm_hour, utc->tm_min, utc->tm_sec);
213 // 成功によるタイマーのキャンセル
214 if (state->ntp_resend_alarm > 0) {
215 cancel_alarm(state->ntp_resend_alarm);
216 state->ntp_resend_alarm = 0;
217 }
218 setRTC(seconds_since_1970);
219 state->status = 1; // 成功
220 } else { // 失敗
221 // printf("invalid ntp response\n");
222 // 失敗によるタイマーのキャンセル
223 if (state->ntp_resend_alarm > 0) {
224 cancel_alarm(state->ntp_resend_alarm);
225 state->ntp_resend_alarm = 0;
226 }
227 state->status = 2; // 成功
228 }
229 pbuf_free(p);
230}

238bool InetAction::setTime()
239{
240 dbgprintf("SNTP ... ");
241
242 // DNSでSNTPサーバーのIPアドレスを取得。最大3回リトライ
243 bool bRet;
244 for (int i = 0; i < 3; i++) {
245 bRet = getHostByName(this->sntp_server_address, 50000);
246 if (bRet) break;
247 for (int j = 0; j < 100; j++) {
248 sleep_ms(10);
249 cyw43_arch_poll();
250 }
251 }
252 if (!bRet) {
253 dbgprintf("DNS Failed.");
254 lastErr_Sntp = -100; // DNSエラーコード
255 return false;
256 }
257 // SNTPサーバーのIPアドレスを取得できたので、NTPのリクエストを送信
258 NTP_T* state = (NTP_T*)calloc(1, sizeof(NTP_T));
259 if (!state) {
260 lastErr_Sntp = -101; // メモリ割り当てエラーコード
261 dbgprintf("SNTP Failed.\nFailed to acllocate state.");
262 return false;
263 }
264 state->ntp_pcb = udp_new_ip_type(IPADDR_TYPE_ANY);
265 if (!state->ntp_pcb) {
266 lastErr_Sntp = -102; // PCB作成エラーコード
267 dbgprintf("SNTP Failed.\nFailed to create PCB.");
268 free(state);
269 return false;
270 }
271 state->ntp_server_address = this->addrSNTP; // DNSで取得したIPアドレスを設定
272
273 // UDP受信コールバックを設定
274 udp_recv(state->ntp_pcb, ntp_recv, state);
275
276 // タイムアウト用アラームをセット
277 state->ntp_resend_alarm = add_alarm_in_ms(NTP_RESEND_TIME, ntp_failed_handler, state, true);
278 state->status = 0;
279 // NTPリクエスト送信
280 cyw43_arch_lwip_begin();
281 struct pbuf* p = pbuf_alloc(PBUF_TRANSPORT, NTP_MSG_LEN, PBUF_RAM);
282 uint8_t* req = (uint8_t*)p->payload;
283 memset(req, 0, NTP_MSG_LEN);
284 req[0] = 0x1b;
285 udp_sendto(state->ntp_pcb, p, &state->ntp_server_address, NTP_PORT);
286 pbuf_free(p);
287 cyw43_arch_lwip_end();
288
289 // 応答待ちループ
290 while (true) {
291 if (state->status != 0) {
292 break;
293 }
294 sleep_ms(100);
295 cyw43_arch_poll();
296 cyw43_arch_wait_for_work_until(state->ntp_test_time);
297 }
298 if (state->status == 1) { // 成功
299 datetime_t dt;
300 bool bRet = rtc_get_datetime(&dt);
301 dbgprintf("Success!\n");
302 time_t now = time(NULL);
303 setlocale(LC_TIME, settings.getLocaleStr());
304 setenv("TZ", settings.getTzStr(), 1);
305 tzset();
306 struct tm* local = localtime(&now);
307
308 dbgprintf("Time: %d-%02d-%02d %02d:%02d:%02d\n",
309 local->tm_year + 1900, local->tm_mon + 1, local->tm_mday,
310 local->tm_hour, local->tm_min, local->tm_sec);
311 return true;
312 } else if (state->status == 2) { // 失敗
313 dbgprintf("SNTP Failed.\nFailed to get Info from SNTP");
314 return false;
315 } else if (state->status == 3) { // タイムアウト
316 dbgprintf("SNTP Failed.\nSNTP Timeout");
317 return false;
318 }
319 dbgprintf("Failed.\nUnknown Error");
320 return false;
321}
