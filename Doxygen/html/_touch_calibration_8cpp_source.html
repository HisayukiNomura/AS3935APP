37bool TouchCalibration::run()
38{
39 while (true) {
40 showMenu();
41 // タッチで上半分/下半分を選択、ロングタップで終了
42 while (m_mode == MODE_SELECT) {
43 if (m_ts->touched()) {
44 TS_Point p = m_ts->getPointOnScreen();
45 if (p.y > 20 && p.y < 120) {
46 m_mode = MODE_CALIBRATE;
47 } else if (p.y > 120 && p.y < 220) {
48 m_mode = MODE_TEST;
49 } else if (p.y > 220) {
50 m_tft->fillScreen(ILI9341_BLACK);
51 m_tft->setTextColor(ILI9341_WHITE);
52 m_tft->setCursor(10, 40);
53 m_tft->printf("終了します\n");
54 if (m_ts->touched()) {};
55 sleep_ms(1000);
56 if (isCalibDone) {
57 if (minX != 0 && maxX != 0xFFFF && minY != 0 && maxY != 0xFFFF) {
58 return true;
59 }
60 }
61 return false;
62 }
63 }
64 }
65 if (m_mode == MODE_CALIBRATE) {
66 runCalibrateMode();
67 m_mode = MODE_SELECT;
68 if (m_ts->touched()) {};
69 } else if (m_mode == MODE_TEST) {
70 runTestMode();
71 m_mode = MODE_SELECT;
72 if (m_ts->touched()) {};
73 } else {
74 break;
75 }
76 }
77 return false;
78}

155void TouchCalibration::runCalibrateMode()
156{
157
158 const char* cornerMsg[4] = {"左上", "右上", "右下", "左下"};
159 minX = 0xFFFF;
160 maxX = 0;
161 minY = 0xFFFF;
162 maxY = 0;
163
164 for (int corner = 0; corner < 4; ++corner) {
165 m_tft->fillScreen(ILI9341_BLUE);
166 m_tft->setTextColor(ILI9341_WHITE);
167 m_tft->setCursor(10, 30);
168 m_tft->printf("%sを10回タップ\n", cornerMsg[corner]);
169 for (int i = 0; i < 10; ++i) {
170 while (!m_ts->touched()) {}
171 TS_Point p = m_ts->getPoint(); // 生のデータ
172 TS_Point pOnScr = m_ts->getPointOnScreen();
173 if (p.x > maxX) maxX = p.x; // 最大値更新
174 if (p.x < minX) minX = p.x; // 最小値更新
175 if (p.y > maxY) maxY = p.y; // 最大値更新
176 if (p.y < minY) minY = p.y; // 最小値更新
177 drawTouchPoint(pOnScr.x, pOnScr.y, ILI9341_YELLOW);
178
179 m_tft->setTextColor(STDCOLOR.WHITE, STDCOLOR.BLACK);
180 m_tft->setCursor(10, 60);
181 m_tft->printf("RAW:%d: X:%4d Y:%4d(\n", i + 1, p.x, p.y);
182 m_tft->printf("MOD:%d: X:%4d Y:%4d(\n", i + 1, pOnScr.x, pOnScr.y);
183 while (m_ts->touched()) {}
184 }
185 m_tft->setCursor(10, 100);
186 m_tft->printf("結果：(%d,%d) - (%d,%d)", minX, minY, maxX, maxY);
187 sleep_ms(1000);
188 }
189 // キャリブレーション値をtsに設定（setCalibrationValueは未実装のためコメントアウト）
190
191 m_ts->setCalibration(minX, minY, maxX, maxY); // TODO: XPT2046_Touchscreen側で実装必要
192 m_tft->fillScreen(STDCOLOR.BLACK);
193 m_tft->setTextColor(STDCOLOR.WHITE, STDCOLOR.BLACK);
194 m_tft->setCursor(10, 60);
195 m_tft->printf("キャリブレーション完了\n");
196 isCalibDone = true;
197 sleep_ms(1000);
198}
