62 // 現在時刻を取得
63 time_t now = time(NULL);
64 struct tm* local = localtime(&now);
65
66 if (isClock24Hour) {
67 uint8_t hourU = local->tm_hour / 10;
68 uint8_t hourL = local->tm_hour % 10;
69 if (prevHourU != hourU) {
70 if (hourU == 0) {
71 pTFT->drawRGBBitmap(x, y, num_blank, 32, 51); // 分の十の位
72 } else {
73 pTFT->drawRGBBitmap(x, y, numTable[local->tm_hour / 10], 32, 51); // 分の十の位
74 }
75 prevHourU = hourU;
76 }
77 if (prevHourL != hourL) {
78 pTFT->drawRGBBitmap(x + 32, y, numTable[local->tm_hour % 10], 32, 51); // 分の一の位
79 prevHourL = hourL;
80 }
81 // コロン
82 pTFT->drawRGBBitmap(x + 72, y, colon, 16, 51);
83 x += 96;
84 } else {
85 uint8_t hour12 = local->tm_hour % 12;
86 if (hour12 == 0) hour12 = 12; // 0時は12時に変換
87 if (prevHour != hour12) {
88 pTFT->drawRGBBitmap(x, y, numTable[hour12], 32, 51);
89 prevHour = hour12;
90 pTFT->drawRGBBitmap(x + 40, y, colon, 16, 51);
91 }
92 x += 64;
93 }
94
95 // 分
96 uint8_t minutesU = local->tm_min / 10;
97 uint8_t minutesL = local->tm_min % 10;
98 if (prevMinutesU != minutesU) {
99 pTFT->drawRGBBitmap(x, y, numTable[local->tm_min / 10], 32, 51);
100 prevMinutesU = minutesU;
101 }
102 if (prevMinutesL != minutesL) {
103 pTFT->drawRGBBitmap(x + 32, y, numTable[local->tm_min % 10], 32, 51);
104 prevMinutesL = minutesL;
105 }
106 x += 70;
107 // AMPM
108 if (isClock24Hour == false) {
109 uint8_t ampm = local->tm_hour / 12;
110 if (prevAmPm != ampm) {
111 if (ampm == 0) {
112 pTFT->drawRGBBitmap(x, y, num_am, 30, 30);
113 } else {
114 pTFT->drawRGBBitmap(x, y, num_pm, 30, 30);
115 }
116 prevAmPm = ampm;
117 }
118 }
119 // 秒
120 if (prevSeconds != local->tm_sec) {
121 int secX;
122 int secY;
123 const uint16_t* pSecPict;
124 uint8_t qIdx = local->tm_sec / 15;
125 if (qIdx == 0) { // ０～１４秒
126 secX = x + 11;

128 pSecPict = secQ1;
129 } else if (qIdx == 1) { // １５～２９秒
130 secX = x + 11;
131 secY = y + 32 + 11;
132 pSecPict = secQ2;
133 } else if (qIdx == 2) { // ３０～４４秒
134 secX = x;
135 secY = y + 32 + 11;
136 pSecPict = secQ3;
137 } else { // ４５～５９秒
138 secX = x;
139 secY = y + 32;
140 pSecPict = secQ4;
141 }
142 if (local->tm_sec == 0) {
143 pTFT->fillRect(x, y + 32, 21, 21, STDCOLOR.SUPERDARK_GRAY);
144 }
145 if (local->tm_sec % 2 == 0 && local->tm_sec % 15 != 14) {
146 pTFT->drawRGBBitmap(secX, secY, secQ0, 10, 10);
147 } else {
148 pTFT->drawRGBBitmap(secX, secY, pSecPict, 10, 10);
