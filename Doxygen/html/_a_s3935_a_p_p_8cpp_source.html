76bool InitDMA()
77{
78
79 // Get a free channel, panic() if there are none
80 // Data will be copied from src to dst
81 const char src[] = "DMA SOURCE BUFFER";
82 char dst[count_of(src)];
83
84 int chan = dma_claim_unused_channel(true);
85
86 // 8 bit transfers. Both read and write address increment after each
87 // transfer (each pointing to a location in src or dst respectively).
88 // No DREQ is selected, so the DMA transfers as fast as it can.
89
90 dma_channel_config c = dma_channel_get_default_config(chan);
91 channel_config_set_transfer_data_size(&c, DMA_SIZE_8);
92 channel_config_set_read_increment(&c, true);
93 channel_config_set_write_increment(&c, true);
94
95 dma_channel_configure(
96 chan, // Channel to be configured
97 &c, // The configuration we just created
98 dst, // The initial write address
99 src, // The initial read address
100 count_of(src), // Number of transfers; in this case each is 1 byte.
101 true // Start immediately.
102 );
103
104 // We could choose to go and do something else whilst the DMA is doing its
105 // thing. In this case the processor has nothing else to do, so we just
106 // wait for the DMA to finish.
107 dma_channel_wait_for_finish_blocking(chan);
108
109 // The DMA has now copied our text from the transmit buffer (src) to the
110 // receive buffer (dst), so we can print it out from there.
111 puts(dst);
112 bool RetVal = (strcmp(src, dst) == 0) ? true : false;
113
114 return RetVal;
115}
