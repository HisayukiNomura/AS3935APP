160uint32_t AS3935::Calibrate()
161{
162 uint32_t frecCnt[0x10] = {0}; // キャパシタごとの周波数カウント
163 uint32_t minDif = 0xFFFFFFFF; // 最小の周波数分周比
164 m_u8calibratedCap = 0; // キャリブレーションされたキャパシタの値
165 // 周波数をカウントしていく
166 for (byte b = 0; b < 0x10; b++) {
167 writeRegAndData_1(REG08_LCO_SRCO_TRCO_CAP, DISPLCO_ON | (TUN_CAP_MASK & b));
168 delay(50);
169 // 指定した秒数（デフォルト１秒）の間、IRQピンの周波数をカウントする
170 frecCnt[b] = FreqCounter::start(m_u8IrqPin, m_timeCalibration);
171 // カウントを1000msに換算しておく
172 if (m_timeCalibration != 1000) {
173 frecCnt[b] = uint32_t(((double)frecCnt[b] / (double)m_timeCalibration) * 1000); // 1秒間のパルス数（周波数）を取得
174 }
175
176 frecCnt[b] = frecCnt[b] * 16; // FDIV_RATIO_1_16 に指定してあるので、周波数は1/16になっている。ここで戻しておく。
177 uint32_t curDif = frecCnt[b] > 500000 ? frecCnt[b] - 500000 : 500000 - frecCnt[b]; // 50MHzを基準にして、どれだけずれているかを計算
178 if (curDif < minDif) {
179 minDif = curDif; // 最小の周波数分周比を更新
180 m_u8calibratedCap = b; // キャリブレーションされたキャパシタの値を更新
181 }
182 // m_pTft->printf("Cap:%3dpF Freq:%4.1fKHz\n", b * 8, (float)frecCnt[b] / 1000);
183 dbgprintf("o");
184 }
185 dbgprintf("\n");
186 return frecCnt[m_u8calibratedCap];
